-- =========================
-- DATABASE
-- =========================
CREATE DATABASE IF NOT EXISTS pizzeria_itconsulting;
USE pizzeria_itconsulting;

-- =========================
-- PIZZA BASE
-- =========================
CREATE TABLE pizza_base (
    id INT PRIMARY KEY AUTO_INCREMENT,
    nome VARCHAR(50) NOT NULL,
    descrizione TEXT,
    prezzo_base DECIMAL(5,2) NOT NULL
);

-- =========================
-- PIZZA (con ingredienti salvati come stringa)
-- =========================
CREATE TABLE pizza (
    id INT PRIMARY KEY AUTO_INCREMENT,
    id_pizza_base INT NOT NULL,
    ingredienti TEXT, -- ← STRINGA tipo: "Mozzarella,Funghi,Salame"
    prezzo_totale DECIMAL(6,2),

    FOREIGN KEY (id_pizza_base) REFERENCES pizza_base(id)
);

-- =========================
-- ORDINE
-- =========================
CREATE TABLE ordine (
    id INT PRIMARY KEY AUTO_INCREMENT,
    id_pizza INT NOT NULL,
    stato VARCHAR(30) NOT NULL,
    data_creazione TIMESTAMP DEFAULT CURRENT_TIMESTAMP,

    FOREIGN KEY (id_pizza) REFERENCES pizza(id) ON DELETE CASCADE
);
